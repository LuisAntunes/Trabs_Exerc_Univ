#####
#Create Turma
#####
create table turma(
nomeT varchar(30),
DataInAula date,
HoraInAula time,
primary key(nomeT,DataInAula,HoraInAula)
);

#####
#Create Professor
#####
Create table professor(
NBIp integer,
NIF integer,
nomeP varchar(30),
morada varchar(30),
TelefoneP integer,
primary key(NBIp, NIF)
);

#####
#Create Alunos
#####
create table alunos(
NBIa integer,
nomeA varchar(30),
TelefoneA integer,
primary key(NBIa)
);

#####
#Create Individual
#####
create table individual(
nomeT varchar(30),
DataInAula date,
HoraInAula time,
TipoInstrumento varchar(30),
primary key(nomeT,DataInAula,HoraInAula)
);

#####
#Create Colectiva
#####
create table colectiva(
nomeT varchar(30),
DataInAula date,
HoraInAula time,
primary key(nomeT,DataInAula,HoraInAula)
);


#####
#Create Regime Livre
#####
create table regime_livre(
NBIa integer,
primary key(NBIa),
foreign key(NBIa) references Alunos on delete restrict
);

#####
#Create Supletivo
#####
create table supletivo(
NBIa integer,
grau integer,
primary key(NBIa),
foreign key(NBIa) references alunos on delete restrict
);

#####
#Create Integrado
#####
create table integrado(
NBIa integer,
grau integer,
primary key(NBIa),
foreign key(NBIa) references Alunos on delete restrict
);

#####
#Create Iniciação
#####
create table iniciacao(
NBIa integer,
nomeA varchar(30),
TelefoneA integer,
primary key(NBIa),
foreign key(NBIa) references Alunos on delete restrict
);

#####
#Create Disciplina
#####
create table disciplina(
trimestre integer,
TipodeAula varchar(30),
nota integer,
primary key(trimestre,TipodeAula)
);


#####
#Create R1
#####
create table r1(
NBIa integer,
nomeT varchar(30),
DataInAula date,
HoraInAula time,
primary key(NBIa,nomeT,DataInAula,HoraInAula),
foreign key(NBIa) references Alunos on delete restrict,
foreign key(nomeT) references Turma on delete restrict,
foreign key(DataInAula) references Turma on delete restrict,
foreign key(HoraInAula) references Turma on delete restrict
);

#####
#Create R2
#####
create table r2(
NBIa integer,
trimestre integer,
TipodeAula varchar(30),
grau integer,
nota integer,
primary key(NBIa,tipodeAula,trimestre,grau),
foreign key(NBIa) references Alunos on delete restrict,
foreign key(tipodeAula) references Disciplina on delete restrict,
foreign key(trimestre) references Disciplina on delete restrict
);

#####
#Create R3
#####
create table r3(
NBIa integer,
trimestre integer,
TipodeAula varchar(30),
grau integer,
nota integer,
primary key(NBIa,trimestre,TipodeAula,grau)
foreign key(NBIa) references Alunos on delete restrict,
foreign key(trimestre) references Disciplina on delete restrict
foreign key(tipodeAula) references Disciplina on delete restrict,
);

#####
#Create R4
#####
create table r4(
NBIa integer,
trimestre integer,
TipodeAula varchar(30),
primary key(NBIa,trimestre,TipodeAula)
foreign key(NBIa) references Alunos on delete restrict,
foreign key(trimestre) references Aula on delete restrict,
foreign key(TipodeAula) references Aula on delete restrict
);

#####
#Create R5
#####
create table r5(
nomeT varchar(30),
DataInAula varchar(30),
HoraInAula varchar(30),
TipodeAula varchar(30),
aulasPrevistas varchar(30),
primary key(NomeT,DataInAula,HoraInAula,tipodeAula,trimestre)
foreign key(nomeT) references Turma on delete restrict,
foreign key(DataInAula) references Turma on delete restrict,
foreign key(HoraInAula) references Turma on delete restrict,
foreign key(tipodeAula) references Aula on delete restrict,
);

#####
#Create R6
#####
create table r6(
NBIa integer,
trimestre integer,
TipodeAula varchar(30),
primary key(NBIa,tipodeAula,trimestre)
foreign key(NBIa) references Alunos on delete restrict,
foreign key(tipodeAula) references Aula on delete restrict,
foreign key(trimestre) references Aula on delete restrict
);

#####
#Create R7
#####
create table r7(
NBIa integer,
nomeT varchar(30),
DataInAula varchar(30),
HoraInAula varchar(30),
primary key(NBIa)
foreign key(NBIa) references Alunos on delete restrict,
foreign key(nomeT) references Turma on delete restrict,
foreign key(DataInAula) references Turma on delete restrict
foreign key(HoraInAula) references Turma on delete restrict,
);

#####
#Create R8
#####
create table r8(
NBIp integer,
NIF integer,
nomeT varchar(30),
DataInAula varchar(30),
HoraInAula varchar(30),
primary key(NBIp,NIF),
foreign key(NBIp) references Professor on delete restrict,
foreign key(NIF) references Professor on delete restrict,
foreign key(nomeT) references Turma on delete restrict,
foreign key(DataInAula) references Turma on delete restrict
foreign key(HoraInAula) references Turma on delete restrict,
);

#####
#Create R9
#####
create table r9(
TipoInstrumento varchar(30),
NBIa integer,
primary key(TipoInstrumento,NBIa),
foreign key(NBIa) references Alunos on delete restrict,
foreign key(TipoInstrumento) references Individual on delete restrict
);

#####
#Create R10
#####
create table r10(
NBIa integer,
trimestre integer,
TipodeAula varchar(30),
presencas integer,
primary key(tipodeAula,trimestre,NBIa)
foreign key(NBIa) references Aluno on delete restrict,
foreign key(trimestre) references Aula on delete restrict,
foreign key(TipodeAula) references Aula on delete restrict,
);

#####
#Create R11
#####
create table r11(
tipodeAula varchar(30),
trimestre integer,
NBIp integer,
NIF integer,
primary key(trimestre,TipodeAula)
foreign key(trimestre) references Aula on delete restrict,
foreign key(TipodeAula) references Aula on delete restrict,
foreign key(NBIp) references Professor on delete restrict,
foreign key(NIF) references Professor on delete restrict
);
